  30 #ifndef _SYS_BUF_H_
  31 #define _SYS_BUF_H_
  32 
  33 #include <sys/types.h>
  34 #include <sys/cdefs.h>
  35 #include <sys/list.h>
  36 
  37 /*
  38  * Buffer header
  39  */
  40 struct buf {
  41         struct list     b_link;         /* link to block list */
  42         int             b_flags;        /* see defines below */
  43         dev_t           b_dev;          /* device number */
  44         int             b_blkno;        /* block # on device */
  45         mutex_t         b_lock;         /* lock for access */
  46         char            *b_data;        /* pointer to data buffer */
  47 };
  48 
  49 /*
  50  * These flags are kept in b_flags.
  51  */
  52 #define B_BUSY          0x00000001      /* I/O in progress. */
  53 #define B_DELWRI        0x00000002      /* delay I/O until buffer reused. */
  54 #define B_INVAL         0x00000004      /* does not contain valid info. */
  55 #define B_READ          0x00000008      /* read buffer. */
  56 #define B_DONE          0x00000010      /* I/O completed. */
  57 
  58 __BEGIN_DECLS
  59 struct buf *getblk(dev_t, int);
  60 int     bread(dev_t, int, struct buf **);
  61 int     bwrite(struct buf *);
  62 void    bdwrite(struct buf *);
  63 void    binval(dev_t);
  64 void    brelse(struct buf *);
  65 void    bflush(struct buf *);
  66 void    bio_sync(void);
  67 void    bio_init(void);
  68 __END_DECLS
  69 
  70 #endif /* !_SYS_BUF_H_ */
