  31  * crt0.S - C startup routine
  32  */
  33 
  34 /**
  35  * Stack layout:
  36  *      argc            argument counter (integer)
  37  *      argv[0]         program name (pointer)
  38  *      argv[1...N]     program args (pointers)
  39  *      argv[argc-1]    end of args (integer)
  40  *      NULL
  41  *      env[0...N]      environment variables (pointers)
  42  *      NULL
  43  */
  44 
  45 /*
  46  * Entry point called by kernel first
  47  */
  48         .text
  49         .global _start
  50 _start:
  51         lwz     r3, 0(r1)
  52         addi    r4, r1, 4
  53         slwi    r5, r3, 2
  54         add     r5, r5, r4
  55         subi    r1, r1, 8       /* adjust stack to call C function */
  56         b       __prex_main
