  30 #ifndef _DEVCTL_H
  31 #define _DEVCTL_H
  32 
  33 /*
  34  * Device control code
  35  */
  36 #define _DEVC(group,num)        (u_long)(((group) << 16) | (num))
  37 
  38 /*
  39  * Power management
  40  */
  41 #define DEVCTL_PM_POWERDOWN     _DEVC('P', 0)   /* power down */
  42 #define DEVCTL_PM_POWERUP       _DEVC('P', 1)   /* power up */
  43 #define DEVCTL_PM_CHGPOLICY     _DEVC('P', 2)   /* change power policy */
  44 #define DEVCTL_PM_LCDOFF        _DEVC('P', 3)   /* lcd off */
  45 #define DEVCTL_PM_LCDON         _DEVC('P', 4)   /* lcd on */
  46 #define DEVCTL_PM_CHGFREQ       _DEVC('P', 5)   /* change cpu frequency */
  47 
  48 /*
  49  * Plug and play
  50  */
  51 #define DEVCTL_PNP_ATTACH       _DEVC('N', 0)   /* device attach */
  52 #define DEVCTL_PNP_DETACH       _DEVC('N', 1)   /* device detach */
  53 #define DEVCTL_PNP_INSERT       _DEVC('N', 2)   /* media insertion */
  54 #define DEVCTL_PNP_REMOVE       _DEVC('N', 3)   /* media removal */
  55 
  56 /*
  57  * Debug
  58  */
  59 #define DEVCTL_DBG_DEVSTAT      _DEVC('D', 0)   /* dump device state */
  60 #define DEVCTL_DBG_ENTERKD      _DEVC('D', 1)   /* entering kernel debugger */
  61 #define DEVCTL_DBG_EXITKD       _DEVC('D', 2)   /* eixt kernel debugger */
  62 
  63 #endif /* !_DEVCTL_H */
