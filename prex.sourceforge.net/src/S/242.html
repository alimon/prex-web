  35 #include <sys/cdefs.h>

  37 #ifdef __lint__
  38 typedef void *                  va_list;
  39 #define va_start(a, l)          (0)
  40 #define va_arg(a, t)            (0)
  41 #define va_end                  /* nothing */
  42 #define va_copy(d, s)           ((d) = (s))
  43 
  44 #elif __GNUC_PREREQ__(2, 96)
  45 #define va_list                 __builtin_va_list
  46 #define va_start(ap, last)      __builtin_va_start((ap), (last))
  47 #define va_arg                  __builtin_va_arg
  48 #define va_end                  __builtin_va_end
  49 #define va_copy(dest, src)      __builtin_va_copy((dest), (src))
  50 #else
  51 
  52 typedef char *                  va_list;
  53 
  54 #define __va_size(type) \
  55         (((sizeof(type) + sizeof(long) - 1) / sizeof(long)) * sizeof(long))
  56 
  57 #define va_start(ap, last) \
  58         ((ap) = (va_list)&(last) + __va_size(last))
  59 
  60 #define va_arg(ap, type) \
  61         (*(type *)(void *)((ap) += __va_size(type), (ap) - __va_size(type)))
  62 
  63 #define va_end(ap)
  64 
  65 #define va_copy(dest, src)      ((dest) = (src))
  66 #endif
  67 
  68 #endif /* !_PPC_STDARG_H */
