  44 #define FILE_NUM 100
  45 
  46 static char path[128];
  47 
  48 int
  49 main(int argc, char *argv[])
  50 {
  51         int fd, i;
  52         char file[128];
  53         pid_t pid = getpid();
  54 
  55         sprintf(path, "t%05d", pid);
  56         if (mkdir(path, 0770) < 0)
  57                 err(1, "mkdir(%s)", path);
  58         if (chdir(path) == -1)
  59                 err(1, "chdir(%s)", path);
  60 
  61         for (i = 0; i < FILE_NUM; i++) {
  62                 sprintf(file, "p%05d.%03d", pid, i);
  63                 if ((fd = creat(file, 0660)) == -1) {
  64                         if (errno != EINTR) {
  65                                 warn("creat(%s)", file);
  66                                 break;
  67                         }
  68                 }
  69                 if (fd != -1 && close(fd) == -1)
  70                         err(2, "close(%d)", i);
  71         }
  72 
  73         for (i--; i >= 0; i--) {
  74                 sprintf(file, "p%05d.%03d", pid, i);
  75                 if (unlink(file) == -1)
  76                         err(3, "unlink(%s)", file);
  77         }
  78 
  79         return (0);
  80 }
