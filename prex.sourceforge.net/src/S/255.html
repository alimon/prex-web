  35 /*
  36  * Define the order of 32-bit words in 64-bit words.
  37  */
  38 #define _QUAD_HIGHWORD 0
  39 #define _QUAD_LOWWORD 1
  40 
  41 /*
  42  * Definitions for byte order, according to byte significance from low
  43  * address to high.
  44  */
  45 #define LITTLE_ENDIAN   1234    /* LSB first: i386, vax */
  46 #define BIG_ENDIAN      4321    /* MSB first: 68000, ibm, net */
  47 #define PDP_ENDIAN      3412    /* LSB first in word, MSW first in long */

  53 #define BYTE_ORDER      _BYTE_ORDER

  60 #define ntohl(x)        (x)
  61 #define ntohs(x)        (x)
  62 #define htonl(x)        (x)
  63 #define htons(x)        (x)

  65 #else   /* LITTLE_ENDIAN || !defined(__lint__) */
