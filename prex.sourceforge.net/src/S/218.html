  37 /*
  38  * Messages for exec object

  43 /*
  44  * Exec message
  45  */
  46 struct exec_msg {
  47         struct msg_header hdr;          /* message header */
  48         char    path[PATH_MAX];         /* program path */
  49         char    cwd[PATH_MAX];          /* current directory */
  50         int     argc;                   /* number of argment string */
  51         int     envc;                   /* number of environment string */
  52         size_t  bufsz;                  /* size of buffer */
  53         char    buf[ARG_MAX];           /* string for arg/env */
  54 };
  55 
  56 /*
  57  * Capability bind message
  58  */
  59 struct bind_msg {
  60         struct msg_header hdr;          /* message header */
  61         char    path[PATH_MAX];         /* program path */
  62 };
  63 
  64 
  65 /* Max size of exec message */
  66 #define MAX_EXECMSG     sizeof(struct exec_msg)
