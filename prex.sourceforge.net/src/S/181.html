  97 /*
  98  * Segment Descriptor
  99  */
 100 struct seg_desc {
 101         u_int limit_lo:16;      /* segment limit (lsb) */
 102         u_int base_lo:16;       /* segment base address (lsb) */
 103         u_int base_mid:8;       /* segment base address (middle) */
 104         u_int type:8;           /* type */
 105         u_int limit_hi:4;       /* segment limit (msb) */
 106         u_int size:4;           /* size */
 107         u_int base_hi:8;        /* segment base address (msb) */
 108 } __packed;
 109 
 110 /*
 111  * Gate Descriptor
 112  */
 113 struct gate_desc {
 114         u_int offset_lo:16;     /* gate offset (lsb) */
 115         u_int selector:16;      /* gate segment selector */
 116         u_int nr_copy:8;        /* stack copy count */
 117         u_int type:8;           /* type */
 118         u_int offset_hi:16;     /* gate offset (msb) */
 119 } __packed;
